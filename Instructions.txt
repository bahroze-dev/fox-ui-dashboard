### Jenkins setup with docker and blue ocean commands.
docker network create jenkins
docker volume create jenkins-docker-certs
docker volume create jenkins-data


docker container run --name jenkins-tutorial --rm --detach -v /var/run/docker.sock:/var/run/docker.sock --volume jenkins-data:/var/jenkins_home --volume jenkins-docker-certs:/certs/client:ro --publish 8080:8080 jenkinsci/blueocean
